ld \
  -d -dp -dc \                  # -d -dp -dc：启用force_common_definition，为可重定位文件中的COMM符号分配空间。
  -no-define-common \           # 启用embedded_relocs，不为COMM符号分配地址。
  --force-exe-suffix \          # 强制生成带有.exe后缀的文件。
  --cref \                      # 生成交叉引用报告。
  --warn-mismatch \             # 警告输入文件不匹配。
  --warn-search-mismatch \      # 警告在库搜索期间尝试打开不兼容的库。
  --check-sections \            # 检查段地址是否重叠。
  --accept-unknown-input-arch \ # 接受未知架构的输入文件。
  --out-implib=liboutput.a \    # 设置要生成的导入库的名称。
  --print-output-format \       # 只在标准输出上打印默认输出。
  --print-memory-usage \        # 显示目标内存使用情况。
  --force-group-allocation \    # 强制解决段组。
  -EB \                         # 设置大端（Big Endian）。 EL 小端
  --interpreter=/lib64/ld-linux-x86-64.so.2 \           # 设置运行时解释器的名称。
  -soname libexample.so \                               # 设置运行时库的名称。
  -rpath /usr/local/lib \                               # 设置运行时库搜索路径。
  -rpath-link /usr/lib \                                # 设置链接时库搜索路径。
  --filter libfilter.so \                               # 设置过滤共享对象的符号表。
  --auxiliary libaux.so \                               # 设置辅助过滤器的共享对象。
  --version-exports-section version1 \                  # 设置导出段的版本符号。
  --default-script=default.ld \                         # 设置默认链接脚本。
  -o outputfile \                                       # 输出文件名
  inputfile1.o inputfile2.o                             # 输入文件
